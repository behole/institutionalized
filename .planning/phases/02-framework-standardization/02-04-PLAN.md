```typescript
const runner = new FrameworkRunner<InputType, ResultType>("framework-name", input);

// Phase 1: Parallel agents

    name: `${agent.role}-${i}`,
    provider,
    model: config.models.agent,
    prompt: buildPrompt(agent, input),
    temperature: config.parameters.temperature,
    maxTokens: 2048,
  }))
);
const parsedResults = responses.map(r => parseJSON<AgentResult>(r.content));

// Phase 2: Synthesis

  "facilitator", provider, config.models.facilitator, synthPrompt, 0.7, 2048
);

// Finalize
const { auditLog } = await runner.finalize(fullResult, "complete");
return { ...fullResult, metadata: { ...metadata, costUSD: auditLog.metadata.totalCost } };
```
